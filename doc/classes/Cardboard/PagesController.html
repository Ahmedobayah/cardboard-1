def update
  @page = Cardboard::Page.find(params[:id])
  fix_new_subparts

  if @page.update_attributes(strong_params[:page])
    flash[:success] = "Your page was updated successfully"
    redirect_to edit_page_path(@page)
  else
    render :edit
  end
end
